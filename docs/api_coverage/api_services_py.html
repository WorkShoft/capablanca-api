139 chess_board.fullmove_number = board.fullmove_number

169def _get_expected_score(player_rating, opponent_rating): 

170 """ 

171 https://wikimedia.org/api/rest_v1/media/math/render/svg/51346e1c65f857c0025647173ae48ddac904adcb 

172 

173 Returns the expected score rounded to two decimals 

174 

175 For instance, a player with 100 less ELO points than their opponent has an expected score 

176 of 0.35993500019711494, which is rounded to 0.36 (a 36% 'chance' of winning) 

177 """ 

178 

179 expected_score = 1 / (1 + 10 ** ((opponent_rating - player_rating) / 400)) 

180 

181 return round(expected_score, 2)
